以下是一个使用 **Bun** 安装 Vue 3 + TypeScript + Vite 项目常用依赖的命令列表，包含详细注释说明：

---

### **📌 核心工具链安装命令**
```bash
# 更新所有依赖包
bun update

# 初始化项目（使用 Vite 官方 Vue + TS 模板）
bun create vite bun-vue-app --template vue-ts
cd bun-vue-app

# 安装生产依赖（注释说明每个包的作用）
bun add \
  axios \                # HTTP 客户端
  pinia \                # 状态管理（推荐替代 Vuex）
  pinia-plugin-persistedstate \    # Pinia 持久化插件
  vue-router \            # 官方路由
  vue-i18n \              # 国际化
  lodash-es \             # lodash 的 es 版本
  nprogress \             # 顶部进度条
  screenfull \            # 全屏控制
  echarts vue-echarts \   # 图表库
  gsap \                  # 动画库
  @vueuse/core            # Vue 组合式工具集（常用 hooks）
  
bun add \
  axios \
  pinia \
  pinia-plugin-persistedstate \
  vue-router \
  vue-i18n \
  lodash-es \
  nprogress \
  screenfull \
  echarts vue-echarts \
  gsap


# 安装开发依赖（带注释的 DevDependencies）
bun add -d \
  @types/bun \                # Bun 环境下的 TypeScript 类型支持
  @types/nprogress \
  @types/lodash-es \
  @biomejs/biome \            # 一体化 Lint + 格式化工具（替代 ESLint/Prettier）
  unocss \                    # 原子化 CSS 引擎
  @unocss/reset \              # Unocss 的默认样式重置
  sass \                      # Sass 预处理器
  postcss postcss-preset-env cssnano \      # PostCSS 插件
  vitest \                    # 测试框架（兼容 Vite）
  @vitest/ui \                # Vitest 的图形化界面
  @intlify/unplugin-vue-i18n \    # 为 vue-i18n 提供的构建工具链插件（开发时优化）
  unplugin-auto-import \      # 自动导入工具
  unplugin-vue-components \   # 自动导入组件工具
  unplugin-icons \            # SVG 图标加载工具
  cz-git                      # Commitizen 规范

bun add -d \
  @types/bun \
  @types/nprogress \
  @biomejs/biome \
  unocss \
  @unocss/reset \
  sass \
  postcss postcss-preset-env \
  vitest \
  @vitest/ui \
  @intlify/unplugin-vue-i18n \
  unplugin-auto-import \
  unplugin-vue-components \
  unplugin-icons \
  cz-git

# vue 常用插件 https://github.com/vitejs/awesome-vite?tab=readme-ov-file#vue
# Vite 社区插件集合 https://github.com/vitejs/awesome-vite#plugins
# https://github.com/vitejs/awesome-vite?tab=readme-ov-file#helpers

# 支持离线能力
# https://vite-pwa-org-zh.netlify.app/
# https://www.npmjs.com/package/vite-plugin-pwa
bun add -d vite-plugin-pwa
# https://github.com/vbenjs/vite-plugin-mock
bun add -d vite-plugin-mock mockjs @types/mockjs
# https://www.npmjs.com/package/vite-plugin-restart
bun add -d vite-plugin-restart
# https://www.npmjs.com/package/vite-plugin-remove-console
bun add -d vite-plugin-remove-console
# https://github.com/svitejs/vite-plugin-qrcode
bun add -d vite-plugin-qrcode
# https://vitepress.dev/zh/
bun add -d vitepress
# https://cn.sli.dev/
bun create slidev
# Vitesse 是一个基于 Vite 的高性能 Vue 3 开发模板
# https://github.com/antfu-collective/vitesse/tree/main
vitesse
# 由 Vite 提供支持的 WebExtension（Chrome、FireFox 等）入门模板。
# https://github.com/antfu-collective/vitesse-webext
vitesse-webext
# HuLa是一款基于Tauri v2+Vue3的桌面即时通讯应用程序（不仅仅是即时通讯），兼容Win、MacOS、Linux、Android、IOS 
# https://github.com/HuLaSpark/HuLa
HuLa
# 将项目捆绑的生产文件上传到阿里 OSS，html 除外
# https://github.com/xiaweiss/vite-plugin-ali-oss
vite-plugin-ali-oss
# https://github.com/cssninjaStudio/unplugin-fonts
unplugin-fonts
# https://github.com/posva/unplugin-vue-router#readme
bun add -d unplugin-vue-router
# https://www.npmjs.com/package/vite-plugin-pages
bun add -d vite-plugin-pages
# https://www.npmjs.com/package/vite-plugin-md
bun add -d vite-plugin-md
# https://www.npmjs.com/package/unplugin-vue-markdown
bun add unplugin-vue-markdown
# https://github.com/jaskang/vite-plugin-markdown-preview?tab=readme-ov-file
vite-plugin-markdown-preview
# https://www.npmjs.com/package/@fortawesome/vue-fontawesome
@fortawesome/vue-fontawesome
@fortawesome/fontawesome-free

# 在入口文件（如 main.ts）引入 Reset 样式
# 推荐（类似 Tailwind 的 Reset）
import '@unocss/reset/tailwind.css'

# husky 是一个用于管理 Git 钩子的工具，它可以帮助我们在特定的 Git 事件（如提交和推送）触发时自动执行一些任务，
# 例如代码检查、格式化、测试等等。因此，husky 可以有效地避免开发人员在提交代码前忘记代码检查、格式化等等操作带来的一系列问题。
# https://typicode.github.io/husky/zh/
bun add -d husky lint-staged commitizen
# init 命令简化了项目中的 husky 设置。它会在 .husky/ 中创建 pre-commit 脚本，并更新 package.json 中的 prepare 脚本。随后可根据你的工作流进行修改。
bun husky init
bun husky add .husky/pre-commit "lint-staged"


# 2. 初始化 biome 配置
bun biome init
# 启动测试的 vitest ui 界面
# 最后，你可以访问 Vitest UI 界面，通过 http://localhost:51204/__vitest__/
vitest --ui
```

---

### **📦 完整安装流程**
```bash
# 步骤 1: 创建项目并进入目录
bun create vite unocss-components --template vue-ts
cd unocss-components
# 安装依赖
bun install
# 启动开发服务器
bun run dev
```
